* @retval: None
* @brief: е‘ЁжңҹжҖ§жөӢйҮҸжЁЎејҸ

static void SHT30_Cycle_Mode(void)
{
    uint8_t temp_array[6] = {0};
    uint16_t temp_uint = 0;
    float temp_float = 0;

    // еҗҜеҠЁе‘ЁжңҹжҖ§жөӢйҮҸ
    IIC_Soft.IIC_Start(&SHT30_IIC);
    // еҶҷж“ҚдҪң

    IIC_Soft.IIC_Write_Byte(&SHT30_IIC, 0x27); // High repeat , mps = 10

    Timer6.SHT30_Measure_Timeout = 0;
    // еҸ‘йҖҒжҺҘж”¶ж•°жҚ®е‘Ҫд»Ө
    do
    {
        if (Timer6.SHT30_Measure_Timeout >= TIMER6_2S) // 2sеҶ…жІЎиҺ·еҸ–еҲ°ж•°жҚ®пјҢйҖҖеҮәзӯүеҫ…
        {
            break;
        }    

        IIC_Soft.IIC_Start(&SHT30_IIC);
        IIC_Soft.IIC_Write_Byte(&SHT30_IIC, SHT30_ADDR & SHT30_Write_CMD);
        // 0xE000жҳҜеҗ‘SHT30еҸ–ж•°жҚ®зҡ„жҢҮд»ӨпјҢдё»жңәеҸ‘йҖҒиҜҘжҢҮд»ӨеҗҺејҖе§ӢиҜ»еҸ–SHT30зҡ„жё©ж№ҝеәҰж•°жҚ®
        IIC_Soft.IIC_Write_Byte(&SHT30_IIC, 0xE0);
        IIC_Soft.IIC_Write_Byte(&SHT30_IIC, 0x00);
        // йҮҚж–°еҸ‘йҖҒиө·е§ӢдҝЎеҸ·пјҢеҫҖSHT30еҸ‘йҖҒең°еқҖеҠ иҜ»еҸ–ж•°жҚ®жҢҮд»Ө
        IIC_Soft.IIC_Start(&SHT30_IIC);
    } while (IIC_Soft.IIC_Write_Byte(&SHT30_IIC, SHT30_ADDR | SHT30_Read_CMD) == NACK); // 0еҲҷйҖҖеҮә

    // ејҖе§ӢжҺҘж”¶жөӢйҮҸж•°жҚ®пјҢе№¶и®Ўз®—
    if (Timer6.SHT30_Measure_Timeout < TIMER6_2S)
    {
        temp_array[0] = IIC_Soft.IIC_Read_Byte(&SHT30_IIC, ACK);
        temp_array[1] = IIC_Soft.IIC_Read_Byte(&SHT30_IIC, ACK);
        temp_array[2] = IIC_Soft.IIC_Read_Byte(&SHT30_IIC, ACK);
        temp_array[3] = IIC_Soft.IIC_Read_Byte(&SHT30_IIC, ACK);
        temp_array[4] = IIC_Soft.IIC_Read_Byte(&SHT30_IIC, ACK);

        IIC_Soft.IIC_Stop(&SHT30_IIC);

        // и®Ўз®—жё©еәҰ,зІҫеәҰ0.1
        if (CRC_8(temp_array, 2) == temp_array[2]) // CRC-8 ж ЎйӘҢ
        {
            //  еҸ–еҮә16дҪҚзҡ„жё©еәҰеҖј
            temp_uint = temp_array[0] * 256 + temp_array[1];
            //  ж №жҚ®жүӢеҶҢе…¬ејҸи®Ўз®—пјҢдёәдәҶзІҫеәҰпјҢи®Ўз®—ж•°еҖје…Ҳ*100
            temp_float = ((float)temp_uint) * 0.267032 - 4500;
            //  еҶҚйҷӨд»Ҙ100пјҢеҫ—еҲ°жӯЈеёёжё©еәҰеҖј
            SHT30.fTemperature = temp_float * 0.01;
        }

        // и®Ўз®—ж№ҝеәҰпјҢзІҫеәҰ1%RH
        if (CRC_8(&temp_array[3], 2) == temp_array[5]) // CRC-8 ж ЎйӘҢ
        {
            // еҸ–еҮә16дҪҚзҡ„ж№ҝеәҰеҖј
            temp_uint = temp_array[3] * 256 + temp_array[4];
            // ж №жҚ®жүӢеҶҢе…¬ејҸи®Ўз®—
            temp_float = ((float)temp_uint) * 0.152590;
            temp_float = temp_float * 0.01;
            // йҷӨд»Ҙ100пјҢеҫ—еҲ°жӯЈеёёж№ҝеәҰеҖј
            SHT30.ucHumidity = (unsigned char)temp_float;
        }
    }
}

/*
* @function: CRC_8
* @param: Crc_ptr -> ж ЎйӘҢж•°жҚ®йҰ–ең°еқҖ LEN -> ж ЎйӘҢж•°жҚ®й•ҝеәҰ
* @retval: None
* @brief: CRC-8ж ЎйӘҢ
*/
static uint8_t CRC_8(uint8_t *Crc_ptr,uint8_t LEN)
{
	uint8_t CRC_Value = 0xFF;
	uint8_t i = 0,j = 0;

	for(i=0; i<LEN; i++)
	{
		CRC_Value ^= *(Crc_ptr+i);
		for(j=0; j<8; j++)
		{
			if(CRC_Value & 0x80)
				CRC_Value = (CRC_Value << 1) ^ 0x31;
			else
				CRC_Value = (CRC_Value << 1);
		}
	}
	return CRC_Value;
}

{
    uint8_t Temp_Array[7] = {0x00};
    uint8_t i = 0, Index = 0;

    HAL_UART_DMAStop(&huart_HMI);   // дёІеҸЈеҒңжӯўDMAжҺҘж”¶

    for (i = 0; i < HMI_Rec_Buffer_LENGTH; i++) // иҜ»еҸ–HMIзј“еӯҳж•°жҚ®пјҢе…ұ7еӯ—иҠӮиө·е§ӢеҖјдёә0x65
    {
        if (0 == Index) // жЈҖжөӢй”®еҖјиө·е§Ӣж•°жҚ®0x65
        {
            if (*(HMI.pucRec_Buffer + i) != 0x65)
            {
                continue;
            }
        }
        Temp_Array[Index] = *(HMI.pucRec_Buffer + i);

        if (Protocol_Data_LEN == Index) // е·ІиҜ»еҸ–7еӯ—иҠӮ
        {
            break;
        }
        Index++;
    }
    HAL_UART_Receive_DMA(&huart1, HMI.pucRec_Buffer, (uint16_t)20); // дёІеҸЈејҖеҗҜDMAжҺҘж”¶
    // еӨ„зҗҶж•°жҚ®
    if (Protocol_Data_LEN == Index)
    {
        // дё»йЎөйқўзҡ„й”®еҖјдҝЎжҒҜ
        if (0x00 == Temp_Array[1])
        {
            // жҺ§д»¶t1еј№иө·дәӢд»¶--ж•°з Ғз®Ў
            if ((0x02 == Temp_Array[2]) && (0x00 == Temp_Array[3]))

            }

		{
			SET_SDA(dev->SDA_GPIOx, dev->SDA_GPIO_Pin);
		}
		else
		{
			CLR_SDA(dev->SDA_GPIOx, dev->SDA_GPIO_Pin);
		}

		// SCLзҪ®й«ҳпјҢдј иҫ“ж•°жҚ®
		SET_SCL(dev->SCL_GPIOx, dev->SCL_GPIO_Pin);
		Public.Public_Delay_us(10);
		
		//еҮҶеӨҮеҸ‘йҖҒдёӢдёҖжҜ”зү№дҪҚ
		WR_Byte <<= 1;

	CLR_SCL(dev->SCL_GPIOx, dev->SCL_GPIO_Pin);

	SET_SDA(dev->SDA_GPIOx, dev->SDA_GPIO_Pin);

	Public.Public_Delay_us(10);

	Public.Public_Delay_us(1);

	return ACK_Rspond;

	// ж•°жҚ®жҢү8дҪҚдј иҫ“пјҢй«ҳдҪҚеңЁеүҚпјҢеҲ©з”ЁforеҫӘзҺҜйҖҗдёӘжҺҘж”¶
	for(i=0; i<8; i++)
	{
		// еҮҶеӨҮжҺҘж”¶дёӢдёҖжҜ”зү№дҪҚ
		RD_Byte <<= 1;
		
		// SCLжё…йӣ¶пјҢд»ҺжңәSDAеҮҶеӨҮж•°жҚ®
		CLR_SCL(dev->SCL_GPIOx, dev->SCL_GPIO_Pin);
		Public.Public_Delay_us(10);
		
		// SCLзҪ®й«ҳпјҢиҺ·еҸ–ж•°жҚ®
		SET_SCL(dev->SCL_GPIOx, dev->SCL_GPIO_Pin);
		Public.Public_Delay_us(10);

		RD_Byte |= READ_SDA(dev->SDA_GPIOx, dev->SDA_GPIO_Pin);		
	}
	
	
	// SCLжё…йӣ¶пјҢдё»жңәеҮҶеӨҮеә”зӯ”дҝЎеҸ·
	CLR_SCL(dev->SCL_GPIOx, dev->SCL_GPIO_Pin);
	Public.Public_Delay_us(1);
	
	// дё»жңәеҸ‘йҖҒеә”зӯ”дҝЎеҸ·	
	if (ACK == ACK_Value)
	{
		CLR_SDA(dev->SDA_GPIOx, dev->SDA_GPIO_Pin);
	}
	else
	{
		SET_SDA(dev->SDA_GPIOx, dev->SDA_GPIO_Pin);
  }	
	Public.Public_Delay_us(1);
	
	SET_SCL(dev->SCL_GPIOx, dev->SCL_GPIO_Pin); 	
	Public.Public_Delay_us(10);
	
  // йҮҠж”ҫSDAж•°жҚ®зәҝ
	// SCLе…Ҳжё…йӣ¶пјҢеҶҚйҮҠж”ҫSDAпјҢйҳІжӯўиҝһз»ӯдј иҫ“ж•°жҚ®ж—¶пјҢд»Һжңәй”ҷе°ҶSDAйҮҠж”ҫдҝЎеҸ·еҪ“жҲҗNACkдҝЎеҸ·
	CLR_SCL(dev->SCL_GPIOx, dev->SCL_GPIO_Pin);
  SET_SDA(dev->SDA_GPIOx, dev->SDA_GPIO_Pin);
	Public.Public_Delay_us(1);

	// иҝ”еӣһж•°жҚ®
	return RD_Byte;	
}

/*
* @function: IIC_Write_Buffer
* @param: dev -> з»“жһ„дҪ“жҢҮй’Ҳ addr -> зӣ®ж Үи®ҫеӨҮең°еқҖ reg -> еҜ„еӯҳеҷЁең°еқҖ len -> иҰҒеҸ‘йҖҒзҡ„ж•°жҚ®й•ҝеәҰ data -> жҢҮеҗ‘ж•°жҚ®зј“еҶІеҢәзҡ„жҢҮй’Ҳ
* @retval: 0 --- жҲҗеҠҹ
* @brief: IICеҶҷдёҖдёІж•°жҚ®
*/
static uint8_t IIC_Write_Buffer(IIC_Data_t *dev, uint8_t addr, uint8_t reg, uint32_t len, uint8_t *data)
{
    uint32_t i;
    
    // ејҖе§ӢдҝЎеҸ·
		IIC_Soft.IIC_Start(dev);
    IIC_Soft.IIC_Write_Byte(dev, addr | 0x00);	// еҶҷж“ҚдҪң
    IIC_Soft.IIC_Write_Byte(dev, reg);	// еҸ‘йҖҒеҜ„еӯҳеҷЁең°еқҖ

    for (i = 0; i < len; i++) 
	 {
        IIC_Soft.IIC_Write_Byte(dev, data[i]);
    }
	 IIC_Soft.IIC_Stop(dev);
		
    return 0;
}

/*
* @function: IIC_Read_Buffer
* @param: dev -> з»“жһ„дҪ“жҢҮй’Ҳ addr -> зӣ®ж Үи®ҫеӨҮең°еқҖ reg -> еҜ„еӯҳеҷЁең°еқҖ len -> иҰҒиҜ»еҸ–зҡ„ж•°жҚ®й•ҝеәҰ data -> жҢҮеҗ‘ж•°жҚ®зј“еҶІеҢәзҡ„жҢҮй’Ҳ
* @retval: 0 --- жҲҗеҠҹ
* @brief: IICиҜ»дёҖдёІж•°жҚ®
*/
static uint8_t IIC_Read_Buffer(IIC_Data_t *dev, uint8_t addr, uint8_t reg, uint32_t len, uint8_t *buf)
{
    // ејҖе§ӢдҝЎеҸ·
		IIC_Soft.IIC_Start(dev);
    IIC_Soft.IIC_Write_Byte(dev, addr | 0x00);
    IIC_Soft.IIC_Write_Byte(dev, reg);
    Public.Public_Delay_us(5);
    
    IIC_Soft.IIC_Start(dev);
    IIC_Soft.IIC_Write_Byte(dev, addr | 0x01);	// иҜ»ж“ҚдҪң

    while(len)
		{
        if (len > 1)
				{
					*buf = IIC_Soft.IIC_Read_Byte(dev, ACK);
				}
				else
				{
					*buf = IIC_Soft.IIC_Read_Byte(dev, NACK);
				}
        buf++;
        len--;
    }
    IIC_Soft.IIC_Stop(dev);
    return 0;
}
